### Using Crispy Forms
Install Crispy Forms

    $ pip install django-crispy-forms

config/settings.py

    INSTALLED_APPS = [ ... 'crispy_forms', ]


templates/signup.html

    {% extends 'theme.html' %}
    {% load crispy_forms_tags %}

    {% block content %}
    <form method="post">
        {% csrf_token %}
        {{ form|crispy }}
        <button type="submit">Sign Up</button>
    </form>
    {% endblock content %}
